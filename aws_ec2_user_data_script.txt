#!/bin/bash
yum update -y
sudo yum remove java-1.7.0-openjdk
sudo yum install wget -y
wget --no-cookies --no-check-certificate --header "Cookie: gpw_e24=http%3A%2F%2Fwww.oracle.com%2F; oraclelicense=accept-securebackup-cookie" "http://download.oracle.com/otn-pub/java/jdk/8u131-b11/d54c1d3a095b4ff2b6607d096fa80163/jdk-8u131-linux-x64.rpm"
sudo rpm -ivh jdk-8u45-linux-x64.rpm -y
echo 'export JAVA_HOME=/usr/java/jdk1.8.0_131' >> ~/.bashrc
cd /opt
wget http://www-eu.apache.org/dist/maven/maven-3/3.3.9/binaries/apache-maven-3.3.9-bin.tar.gz
sudo tar xzf apache-maven-3.3.9-bin.tar.gz
sudo ln -s apache-maven-3.3.9 maven
echo 'export M2_HOME=/opt/apache-maven-3.3.9' >> ~/.bashrc
echo 'export PATH=${M2_HOME}/bin:${PATH}' >> ~/.bashrc
wget http://mirror.metrocast.net/apache/tomcat/tomcat-7/v7.0.79/bin/apache-tomcat-7.0.79.tar.gz
sudo tar xzf apache-tomcat-7.0.79.tar.gz
sudo /opt/apache-tomcat-7.0.79/bin/startup.sh
#service httpd start
#chkconfig httpd on
